[docs]    def clear(self):
        f = self.f
        try:
            f.func
        except AttributeError:
            consume_args = 0
        else:
            consume_args = len(f.args)
            f = f.func
        args = f.arguments[consume_args:]
        if f.__defaults__:
            raise NotImplementedError('default argument value handling is not yet implemented')
        self.args = dict({arg : None for arg in args})

    

[docs]    def accept(self, entity):
        try:
            entity.ally_group
        except AttributeError:
            is_sided = False
        else:
            is_sided = True
        kind = None
        if is_sided:
            if entity.is_ally(self.owner):
                kind = 'ally'
            elif entity.is_enemy(self.owner):
                kind = 'enemy'
        if kind is None:
            return False
        matching_args = sorted(list([name for name in self.args if not self.args[name] and name.find(kind) > -1]))
        if not matching_args:
            return False
        arg = matching_args[0]
        self.args[arg] = (entity,)
        return True

    

[docs]    def ready(self):
        return not [None for name in self.args if not self.args[name]]

    

[docs]    def get(self):
        kwargs = dict({arg:self.args[arg][0] for arg in self.args})
        return self.f(**kwargs)

[docs]    def ui_action(self, tp, f):
        """Mark the action as available to UI"""
        if not tp in self._ui_hints:
            self._ui_hints[tp] = dict()
        try:
            name = f.__name__
        except AttributeError:
            name = f.func.__name__
        self._ui_hints[tp][name] = f

    
    @unbound

[docs]    def ui_hints(self, action_type):
        actions = set(self._ui_hints.get(action_type, dict()).values())
        attrs = [getattr(self, nm) for nm in dir(self)]
        actions.update([attr for attr in attrs if callable(attr) and hasattr(attr, 'category') and attr.category == action_type])
        try:
            return [UniversalBattleAction(self, f) for f in actions]
        except KeyError:
            return []

[docs]    def start(self):
        """Start manager: this simply starts new turn for user right now.
        
        (this can mean allocating AP for example)
        """
        self.turnman.play_forever()
        self.turnman.turn()

    

[docs]    def end_turn(self):
        """End turn.
        
        This marks player's turn as being over and processes AI turns
        """
        if isinstance(self.active_controller(), UserController):
            self.active_controller().end_turn()
        # since we use turnman.play_forever(), we only need to do what is
        # already planned
        # NOTE: might end up in non-stop loop if there's no user
        self.turnman.planned()

    

[docs]    def can_finish(self):
        return isinstance(self.turnman.world.state, battle.Finished)

    

[docs]    def end_encounter(self):
        self.turnman.finish()
        self.done = True
        for f in self.post_encounter_hooks:
            f()

    

[docs]    def get_actions(self, entity, action_type):
        if entity == self.selected:
            try:
                entity.ui_hints
            except AttributeError:
                # no UIHints provided..
                return []
            else:
                return entity.ui_hints(action_type)
        return []

    

[docs]    def select_action(self, action):
        self.selected_action = action
        self.selected_action.clear()
        if self.selected_action.ready():
            self.do_action(self.selected_action.get())

    

[docs]    def active_controller(self):
        return self.turnman.next_side()

    

[docs]    def clicked(self, side, entity):
        """Process simple click on entity.
        
        Right now, it selects/"heals" player entity and attacks enemy entity
        """
        if self.selected:
            if self.selected == entity:
                self.deselect()
            elif self.selected_action:
                if self.selected_action.accept(entity):
                    if self.selected_action.ready():
                        self.do_action(self.selected_action.get())
        else:
            if side is self.active_controller().entity:
                self.selected = entity

    

[docs]    def do_action(self, action):
        """Tell turnman that user wants to perform action"""
        self.active_controller().do_action(action)
        self.turnman.planned()
