[docs]    def next_side(self):
        """Returns next/current active side or None if no sides"""
        if not self.sides:
            return None
        if self.queue:
            return self.queue[0]
        else:
            return self.sides[0]

    

[docs]    def new_round(self):
        """Called when new round starts"""
        self.queue = copy.copy(self.sides)
        try:
            self.world.new_round
        except AttributeError:
            pass
        else:
            self.world.new_round()
        try:
            self.world.new_turn
        except AttributeError:
            pass
        else:
            self.world.new_turn()

            self.world.new_turn()

[docs]    def play_forever(self):
        """Continous battle mode where next turn is performed when it can.
        
        This might lead to lock if there are no external pauses (e.g. some
        controller require user input or time to pass).
        """
        self._turns_planned = float('inf')

    

[docs]    def turn(self):
        """Advance till turn ends.
        
        NOTE: if actions required to end turn cannot be performed now, they'll
        be performed as soon as possible (call planned_actions() to try
        perform such delayed actions)
        
        NOTE: should be called only once for a turn. To finish uncompleted turn,
        use planned_actions() instead.
        
        Returns whatever the last step call returned (None if turn is over,
        False if there are still unperformed actions left)
        """
        if self.planned_actions():
            self._turns_planned += 1
            return False
        
        side = self.next_side()
        if side is None:
            r = None
        else:
            if not self.queue:
                self.new_round()
            
            r = True
            while r:
                r = self.step(side)
            if r is None:
                self.new_turn()
            else:
                self._turns_planned += 1
        if r is None:
            self.planned_turns()
        return r

    

[docs]    def step(self, side):
        """Process one step of given side.
        
        Returns True if action was performed and there are more, False if
        no action could be performed, None if side's turn is finished.
        """
        r = side.act()
        if r:
            # action is present
            ar = self.process(r)
            if not ar:
                self.plan_action(r)
                return False
            return True
        # if not r, r is either False or None
        return r

    

[docs]    def plan_action(self, a):
        self._planned.append(a)

    

[docs]    def planned_actions(self):
        """Perform all planned actions.
        
        Return True if there are some actions left, False otherwise
        """
        if self._planned:
            ar = True
            while ar and self._planned:
                next_action = self._planned[0]
                ar = self.process(next_action)
                if ar:
                    self._planned.pop(0)
            if not ar:
                return True
        return False

    

[docs]    def planned_turns(self):
        if self._turns_planned > 0:
            self._turns_planned -= 1
            r = self.turn()
            if r is None:
                return False
            else:
                return True
        return False

    

[docs]    def planned(self):
        self.planned_actions()
        self.planned_turns()



[docs]class LockableTurnman(Turnman):
    """Turnman that can be locked and then actions are not performed.
    
    This is useful for performing various effects when actions happen.
    """
    def __init__(self, *args, **kwargs):
        super(LockableTurnman, self).__init__(*args, **kwargs)
        self._locked = 0
    

[docs]    def lock(self):
        """Add lock."""
        self._locked += 1

    

[docs]    def unlock(self):
        """Release lock. This will also perform planned actions if fully unlocked"""
        self._locked -= 1
        if self._locked < 0:
            raise RuntimeError('too much unlocking')
        if self._locked == 0:
            self.planned()

    

[docs]    def process(self, a):
        if self._locked > 0:
            return False
        return super(LockableTurnman, self).process(a)
