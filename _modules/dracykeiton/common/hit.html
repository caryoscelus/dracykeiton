[docs]    def try_hit(self, enemy):
        try:
            evasion = enemy.evasion
        except AttributeError:
            evasion = -1
        accuracy = self.accuracy
        return accuracy >= evasion

    
    @unbound

[docs]    def force_hit(self, enemy):
        killed = enemy.hurt(self.hit_damage)
        if killed:
            self.killed(enemy)
