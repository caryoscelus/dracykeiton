    @unbound

[docs]    def spend_ap(self, ap):
        """This can be used by action checker.
        
        Returns True if there was enough AP and it's spent, False otherwise.
        """
        if not self.can_act:
            return False
        if self.ap < ap:
            return False
        else:
            self.ap -= ap
            return True
