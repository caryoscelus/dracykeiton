        if self.living == 'unborn':

            raise TypeError('dead cannot be born, try some black magic instead!')
        elif self.living == 'alive':

[docs]    def hurt(self, damage):
        was_alive = self.living == 'alive'
        self.hp -= damage
        if was_alive and self.living == 'dead':
            return True
        return False
